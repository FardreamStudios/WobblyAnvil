// ============================================================
// Cloudflare Worker — Wobbly Anvil Fairy LLM Proxy
//
// Receives requests from the game client, forwards to
// Anthropic Messages API, returns fairy dialogue.
//
// DEPLOYMENT:
//   1. Create a Cloudflare account + Workers project
//   2. Set secret: wrangler secret put ANTHROPIC_API_KEY
//   3. Deploy: wrangler deploy
//
// ENVIRONMENT VARIABLES (Cloudflare dashboard or wrangler.toml):
//   ANTHROPIC_API_KEY — your Anthropic API key (secret)
//   ALLOWED_ORIGIN    — your GitHub Pages URL (optional CORS lock)
//
// REQUEST FORMAT:
//   POST /
//   { type: "reactive"|"chat", system, state, message?, history?, model? }
//
// RESPONSE FORMAT:
//   { line: "fairy dialogue here" }
// ============================================================

var DEFAULT_MODEL = "claude-haiku-4-5-20251001";
var MAX_TOKENS = 60;  // fairy speaks in short lines
var ANTHROPIC_API_URL = "https://api.anthropic.com/v1/messages";

export default {
    async fetch(request, env) {
        // --- CORS ---
        if (request.method === "OPTIONS") {
            return new Response(null, { headers: corsHeaders(env) });
        }

        if (request.method !== "POST") {
            return jsonResponse({ error: "POST only" }, 405, env);
        }

        try {
            var body = await request.json();
            var type = body.type || "reactive";
            var system = body.system || "";
            var state = body.state || {};
            var model = body.model || DEFAULT_MODEL;

            // Build system prompt with game state context
            var fullSystem = system + "\n\nCURRENT GAME STATE:\n" + JSON.stringify(state);

            var messages = [];

            if (type === "chat") {
                // Multi-turn: build conversation from history
                var history = body.history || [];
                for (var i = 0; i < history.length; i++) {
                    var entry = history[i];
                    if (entry.role === "user") {
                        messages.push({ role: "user", content: entry.text });
                    } else if (entry.role === "fairy") {
                        messages.push({ role: "assistant", content: entry.text });
                    }
                }
                // Add current message
                if (body.message) {
                    messages.push({ role: "user", content: body.message });
                }
            } else {
                // Reactive: single turn
                messages.push({
                    role: "user",
                    content: "React to the current game state. Give one short fairy dialogue line.",
                });
            }

            // Ensure messages array is valid (non-empty, starts with user)
            if (messages.length === 0) {
                messages.push({ role: "user", content: "Say something in character." });
            }

            // Call Anthropic
            var apiResponse = await fetch(ANTHROPIC_API_URL, {
                method: "POST",
                headers: {
                    "Content-Type": "application/json",
                    "x-api-key": env.ANTHROPIC_API_KEY,
                    "anthropic-version": "2023-06-01",
                },
                body: JSON.stringify({
                    model: model,
                    max_tokens: MAX_TOKENS,
                    system: fullSystem,
                    messages: messages,
                }),
            });

            if (!apiResponse.ok) {
                var errText = await apiResponse.text();
                console.error("[Worker] Anthropic error:", apiResponse.status, errText);
                return jsonResponse({ line: null, error: "api_error" }, 502, env);
            }

            var data = await apiResponse.json();

            // Extract text from response
            var line = "";
            if (data.content && data.content.length > 0) {
                for (var j = 0; j < data.content.length; j++) {
                    if (data.content[j].type === "text") {
                        line = data.content[j].text;
                        break;
                    }
                }
            }

            return jsonResponse({ line: line }, 200, env);

        } catch (err) {
            console.error("[Worker] Error:", err.message);
            return jsonResponse({ line: null, error: "internal" }, 500, env);
        }
    },
};

function corsHeaders(env) {
    var origin = (env && env.ALLOWED_ORIGIN) || "*";
    return {
        "Access-Control-Allow-Origin": origin,
        "Access-Control-Allow-Methods": "POST, OPTIONS",
        "Access-Control-Allow-Headers": "Content-Type",
        "Access-Control-Max-Age": "86400",
    };
}

function jsonResponse(obj, status, env) {
    return new Response(JSON.stringify(obj), {
        status: status,
        headers: Object.assign({
            "Content-Type": "application/json",
        }, corsHeaders(env)),
    });
}
